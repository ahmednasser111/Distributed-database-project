create database portsaid;
create database cairo_replication;

-------------------------------------------------------------------------------

USE portsaid;

CREATE TABLE products (
    id INT PRIMARY KEY,
    p_name VARCHAR(255),
    price DECIMAL(10, 2),
    sex VARCHAR(10)
);

CREATE TABLE inventory (
    pid INT PRIMARY KEY,
    quantity INT,
    FOREIGN KEY (pid) REFERENCES products(id)
);
CREATE TABLE transactions (
    pid INT,
    quantity INT,
    p_state BIT,
    p_timestamp TIMESTAMP,
    PRIMARY KEY (pid, p_timestamp),
    FOREIGN KEY (pid) REFERENCES products(id)
);


use cairo_replication;

CREATE TABLE inventory (
    pid INT PRIMARY KEY,
    quantity INT
);

CREATE TABLE transactions (
    pid INT,
    quantity INT,
    p_state BIT,
    p_timestamp TIMESTAMP,
    PRIMARY KEY (pid, p_timestamp),
);
